\section*{\textcolor{aegisblue}{\faUser} THE THREE AGENTS}

\agentbox{CONSERVATOR}{Risk-Averse}{Waits for sustained critical issues. Acts only at 95\%+ with 5+ min duration. High confidence in waiting.}{conservator}

\vspace{3pt}
\agentbox{PREVENTER}{Proactive}{Prevents problems before they occur. Monitors velocity and trends. Early intervention at 80\%+.}{preventer}

\section*{\textcolor{aegisblue}{\faLightbulb} BEST PRACTICES}

\textbf{DO:}
\begin{itemize}[leftmargin=12pt,itemsep=1pt]
\item Check \cmd{aegis forecast} before long tasks
\item Let auto-actions run (they learn!)
\item Use \cmd{aegis leaks} when RAM high
\item Disable auto before critical work
\end{itemize}

\textbf{DON'T:}
\begin{itemize}[leftmargin=12pt,itemsep=1pt]
\item Panic if agents disagree (consensus protects you)
\item Ignore critical forecasts (usually correct)
\item Always disable auto (prevents learning)
\item Kill processes blindly (check \cmd{aegis leaks} first)
\end{itemize}
